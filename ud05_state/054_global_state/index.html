    import { Injectable, signal, computed } from '@angular/core';

@Injectable({
    providedIn: 'root'
})
export class GlobalService {
    // Estado global: usuario autenticado
    readonly authenticatedUser = signal<string | null>(null);

    readonly theme = signal<'light' | 'dark'>('light');

    constructor() {

        this.authenticatedUser.set(user);
    }

    setAuthenticatedUser(user: string | null) {
        this.authenticatedUser.set(user);

        }

    private loadUserFromLocalStorage(): string | null {
        return localStorage.getItem('authenticatedUser');
    }

    private saveThemeToLocalStorage(theme: 'light' | 'dark') {
        localStorage.setItem('theme', theme);

    private loadThemeFromLocalStorage(): 'light' | 'dark' {
