@Pipe({
  name: 'productFilter'
})

  transform(products: Product[], filterBy: string): Product[] {
    if (filterBy) {
      // aplica el m√©todo filter de javascript sobre un array
      // y devuelve un nuevo array con los elementos que cumplan la condici√≥n
      //
      // filter toma como par√°metro una funci√≥n an√≥nima que devuelve un booleano
